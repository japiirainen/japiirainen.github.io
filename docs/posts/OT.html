  ins : Char → ℕ → Op
  del : ℕ → Op

𝐓 : Op → Op → Op
𝐓 (ins c₀ i₀) (ins c₁ i₁) = ins 'd' 3
𝐓 (ins c₀ i₀) (del i₁) = ins 'd' 3
𝐓 (del i₀) (ins c₁ i₁) = ins 'd' 3
𝐓 (del i₀) (del i₁) = ins 'd' 3
